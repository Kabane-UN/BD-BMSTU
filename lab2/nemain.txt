package main

import (
	"errors"
	"fmt"
	"math/rand"
	"sync"
	"time"
)

// Типы блокировок
type LockType int

const (
	Shared LockType = iota
	Exclusive
)

// Сообщения
type Request struct {
	TxID    string
	Key     string
	Value   string
	Type    string // "PREPARE", "COMMIT", "ROLLBACK", "READ"
	ReplyCh chan<- Response
}

type Response struct {
	OK    bool
	Value string
	Err   error
}

// Узел распределенной системы
type Node struct {
	id          int
	data        map[string]string
	locks       map[string]string // key (что занято) -> TxID (кем занято)
	pendingData map[string]string // TxID+key -> value (буфер до коммита)
	inbox       chan Request
}

func NewNode(id int) *Node {
	n := &Node{
		id:          id,
		data:        make(map[string]string),
		locks:       make(map[string]string),
		pendingData: make(map[string]string),
		inbox:       make(chan Request, 10),
	}
	go n.run()
	return n
}

// Задержка сети
func networkDelay() {
	time.Sleep(time.Duration(10+rand.Intn(90)) * time.Millisecond)
}

func (n *Node) run() {
	for req := range n.inbox {
		networkDelay()

		switch req.Type {
		case "PREPARE":
			// 2PL: Проверяем блокировку. Если занято другой транзакцией - ABORT
			if currentTx, occupied := n.locks[req.Key]; occupied && currentTx != req.TxID {
				req.ReplyCh <- Response{OK: false, Err: errors.New("lock conflict")}
			} else {
				n.locks[req.Key] = req.TxID
				n.pendingData[req.TxID+req.Key] = req.Value
				req.ReplyCh <- Response{OK: true}
			}

		case "COMMIT":
			// Применяем данные из буфера в основное хранилище
			if val, ok := n.pendingData[req.TxID+req.Key]; ok {
				n.data[req.Key] = val
			}
			// Освобождаем лок и буфер
			delete(n.locks, req.Key)
			delete(n.pendingData, req.TxID+req.Key)
			req.ReplyCh <- Response{OK: true}
		case "READ":
			// 2PL: Если есть лок на ключь, читать нельзя - аборт
			if owner, busy := n.locks[req.Key]; busy && owner != req.TxID {
				req.ReplyCh <- Response{OK: false, Err: errors.New("read-lock conflict")}
			} else {
				// Если лока нет, читаем нормально
				val := n.data[req.Key]
				req.ReplyCh <- Response{OK: true, Value: val}
			}

		case "ROLLBACK":
			// Освобождаем лок и буфер
			delete(n.locks, req.Key)
			delete(n.pendingData, req.TxID+req.Key)
			req.ReplyCh <- Response{OK: true}
		}

	}
}

// Менеджер
type Coordinator struct {
	nodes []*Node
}

func (c *Coordinator) Execute(txID string, writes map[int]map[string]string) error {
	// 1. Phase 1: PREPARE
	fmt.Printf("[Tx %s] Starting Phase 1 (Prepare)\n", txID)
	for nodeIdx, keys := range writes {
		for key, val := range keys {
			replyCh := make(chan Response)
			// Опрашиваем узел
			c.nodes[nodeIdx].inbox <- Request{
				TxID: txID, Key: key, Value: val, Type: "PREPARE", ReplyCh: replyCh,
			}
			resp := <-replyCh
			// Откат, если неудача
			if !resp.OK {
				fmt.Printf("[Tx %s] Conflict on Node %d, key %s. Rolling back...\n", txID, nodeIdx, key)
				c.rollback(txID, writes)
				return resp.Err
			}
		}
	}

	// 2. Phase 2: COMMIT
	fmt.Printf("[Tx %s] All nodes prepared. Starting Phase 2 (Commit)\n", txID)
	for nodeIdx, keys := range writes {
		for key := range keys {
			replyCh := make(chan Response)
			c.nodes[nodeIdx].inbox <- Request{
				TxID: txID, Key: key, Type: "COMMIT", ReplyCh: replyCh,
			}
			<-replyCh
		}
	}
	return nil
}

// Сигналим откат для всех нод
func (c *Coordinator) rollback(txID string, writes map[int]map[string]string) {
	for nodeIdx, keys := range writes {
		for key := range keys {
			replyCh := make(chan Response)
			c.nodes[nodeIdx].inbox <- Request{
				TxID: txID, Key: key, Type: "ROLLBACK", ReplyCh: replyCh,
			}
			<-replyCh
		}
	}
}

func main() {
	// Сценарий: Конфликт транзакций
	// Tx1 занимает ключ, Tx2 пытается его взять во время Prepare
	fmt.Println("--- Test 1: Basic Conflict ---")
	nodes := []*Node{NewNode(0), NewNode(1), NewNode(2)}
	coord := &Coordinator{nodes: nodes}

	wg := sync.WaitGroup{}

	wg.Go(func() {
		// Транзакция 1: пишет в Node 0, key "balance"
		err := coord.Execute("TX-1", map[int]map[string]string{
			0: {"balance": "100"},
		})
		if err == nil {
			fmt.Println("TX-1 success")
		}
	})

	time.Sleep(30 * time.Millisecond) // Даем TX-1 успеть захватить лок

	wg.Go(func() {
		// Транзакция 2: пытается писать в тот же ключ на Node 0
		err := coord.Execute("TX-2", map[int]map[string]string{
			0: {"balance": "200"},
		})
		if err != nil {
			fmt.Printf("TX-2 failed as expected: %v\n", err)
		}
	})

	wg.Wait()
	TestMultiNodeRollback()
	TestReadWriteConflict()
	TestParallelIndependent()
}

func TestMultiNodeRollback() {
	// Тоже самое, но нод несколько (частичное пересечение)

	nodes := []*Node{NewNode(0), NewNode(1), NewNode(2)}
	coord := &Coordinator{nodes: nodes}
	fmt.Println("\n--- Running Test: Multi-Node Atomic Rollback ---")

	// 1. TX-X захватывает ключ на Узле 2
	wg := sync.WaitGroup{}
	wg.Go(func() {
		coord.Execute("TX-X", map[int]map[string]string{
			2: {"storage": "busy"},
		})
	})
	time.Sleep(10 * time.Millisecond)

	// 2. TX-Y пытается обновить Узел 0, 1 и 2.
	// На Узле 2 она встретит конфликт.
	err := coord.Execute("TX-Y", map[int]map[string]string{
		2: {"storage": "update-attempt"},
		0: {"data": "new-0"},
		1: {"data": "new-1"},
	})

	wg.Wait()

	if err != nil {
		fmt.Println("TX-Y aborted as expected. Checking consistency...")
		fmt.Printf("Node 0 data (should be empty): '%s'\n", nodes[0].data["data"])
		fmt.Printf("Node 1 data (should be empty): '%s'\n", nodes[1].data["data"])
		fmt.Printf("Node 2 storage (not empty): '%s'\n", nodes[2].data["storage"])
	}
}

func TestReadWriteConflict() {
	// Read uncommited
	nodes := []*Node{NewNode(0), NewNode(1), NewNode(2)}
	coord := &Coordinator{nodes: nodes}
	fmt.Println("\n--- Running Test: Read-Write Conflict ---")

	// TX-1 начинает запись
	go coord.Execute("TX-WRITE", map[int]map[string]string{
		0: {"config": "v2"},
	})

	time.Sleep(20 * time.Millisecond)

	// TX-READ читает тот же ключ
	reply := make(chan Response)
	nodes[0].inbox <- Request{TxID: "TX-READ", Type: "READ", Key: "config", ReplyCh: reply}
	res := <-reply

	if !res.OK {
		fmt.Println("TX-READ was rejected because TX-WRITE is in progress. Correct!")
	}
}

func TestParallelIndependent() {
	// Нормальное независимое изменение
	nodes := []*Node{NewNode(0), NewNode(1)}
	coord := &Coordinator{nodes: nodes}
	fmt.Println("\n--- Running Test: Parallel Independent Transactions ---")
	start := time.Now()
	wg := sync.WaitGroup{}

	// Эти транзакции работают с разными ключами
	wg.Go(func() {
		coord.Execute("TX-A", map[int]map[string]string{0: {"key1": "val1"}})
		
	})
	wg.Go(func() {
		coord.Execute("TX-B", map[int]map[string]string{0: {"key2": "val2"}})
	})

	wg.Wait()
	fmt.Printf("Both finished in %v (should be fast, ~100-200ms)\n", time.Since(start))
}
