CREATE STREAM nasa_logs_raw (
    host VARCHAR,
    timestamp VARCHAR,
    method VARCHAR,
    url VARCHAR,
    protocol VARCHAR,
    status INT,
    bytes INT
) WITH (
    KAFKA_TOPIC='nasa_access_logs', 
    VALUE_FORMAT='JSON'
);

CREATE STREAM nasa_html_only AS
SELECT 
    url,
    status,
    CASE 
        WHEN status = 200 THEN 1 
        ELSE 0 
    END AS is_success,
    CASE 
        WHEN status >= 400 THEN 1 
        ELSE 0 
    END AS is_error
FROM nasa_logs_raw
WHERE url LIKE '%.html%'; -- Берем .html


CREATE TABLE url_metrics AS
SELECT 
    url,
    COUNT(*) AS total_requests,
    SUM(is_success) AS success_count,
    SUM(is_error) AS error_count,
    (CAST(SUM(is_success) AS DOUBLE) / COUNT(*)) * 100 AS success_rate_percent
FROM nasa_html_only
GROUP BY url
EMIT CHANGES;

    "url" TEXT PRIMARY KEY,
    "total_requests" BIGINT,
    "success_count" BIGINT,
    "error_count" BIGINT,
    "success_rate_percent" DOUBLE PRECISION
);
